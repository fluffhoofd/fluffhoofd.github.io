python socket client 
import socket
s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
s.connect((127.0.0.1, 80))
print(s.recv(1024))
s.close
python socket server 
import socket
s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
s.connect((127.0.0.1, 80))
s.listen(1)
while True:
    c, a = s.accept()
    c.send('message')
    c.close()
